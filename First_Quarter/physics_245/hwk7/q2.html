import numpy as np
from qutip import *

def H_gen(omega, g, N=15):
    """
    Generate Jaynes-Cummings Hamiltonian for given parameters

    Parameters:
        omega(float): qubit frequency
        g(float): coupling strength
        N(int): hilbert space dimention for oscillator

    Returns:
         H(Qobj): the Jaynes-Cummings Hamiltonian in quesion

    """
    a = tensor(qeye(2), qeye(2), destroy(N))
    ad = tensor(qeye(2), qeye(2), destroy(N).dag())
    sz1 = tensor(sigmaz(),qeye(2), qeye(N))
    sz2 = tensor(qeye(2), sigmaz(), qeye(N))
    sp1 = tensor(sigmap(),qeye(2), qeye(N))
    sp2 = tensor(qeye(2), sigmap(), qeye(N))
    sm1 = tensor(sigmam(), qeye(2), qeye(N))
    sm2 = tensor(qeye(2), sigmam(), qeye(N))

    H = omega / 2 * (sz1 + sz2) + \
    2 * omega * (ad * a +1/2) + \
    g / 2 * ((sp1 + sp2) @ a + (sm1 + sm2) @ ad)

    return H


def extract_qubit_states(result):
    """
    Extract the state of the two qubits from the mesolve result
    
    Parameters:
        result(Qobj): the output of mesolver

    Returns:
        tuple:[
        [0] (np.array()): the time evolution of qubit 1
        [1] (np.array()): the time evolution of qubit 2
        ]
    """
    qubit1_state = []
    qubit2_state = []
    for state in result.states:
        
        # partial trace over the oscillator to get the two-qubit state only
        rho = ket2dm(state)
        rho_qubit = Qobj(rho.ptrace([0, 1]))

        # extract the probabilities of the four qubit states using .full().flatten[i]
        p00 = abs(rho_qubit.full().flatten()[0])**2
        p01 = abs(rho_qubit.full().flatten()[1])**2
        p10 = abs(rho_qubit.full().flatten()[2])**2

        qubit1_state.append(p00 + p01)
        qubit2_state.append(p00 + p10)

    return np.array(qubit1_state), np.array(qubit2_state)

def plot_qubit_states(tlist, qubit1_state, qubit2_state, title):
    """
    Plot the state of the two qubits
    
    Parameters:
        tlist(list): time sequence of the corresponding state evolution
        qubit1_state(np.array): qubit 1's time evolution
        qubit2_state(np.array): qubit 2's time evolution
        title(string): title for the plot

    Return:
        plot shown
    """
    plt.figure(figsize=(10, 6))
    plt.plot(tlist, qubit1_state, 'r--', label='Qubit 1 |0⟩')
    plt.plot(tlist, qubit2_state, 'b-', label='Qubit 2 |0⟩')
    plt.xlabel('Time')
    plt.ylabel('Probability')
    plt.title(title)
    # plt.ylim([-0.1, 1.1])
    plt.legend()
    plt.grid(True)
    plt.show()

def U_gen_2(t, g=np.pi * 2 / 100, delta=2 * np.pi):
    """
    Generate unitary time evolution operator after RWA for given parameters

    Parameters:
        t(float): a certain given time
        g(float): coupling strength
        delta(float): detuning

    Returns:
        (np.array): the corresponding unitary evolution operator
    """
    return np.array([[1, 0, 0, 0], 
                     [0, np.cos(g**2 * t / (4 * delta)), 1j * np.sin(g**2 * t / (4 * delta)), 0],
                     [0, 1j * np.sin(g**2 * t / (4 * delta)), np.cos(g**2 * t / (4 * delta)), 0],
                     [0, 0, 0, 1]])
