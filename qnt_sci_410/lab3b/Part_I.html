import numpy as np
from matplotlib import pyplot as plt
from matplotlib.ticker import MaxNLocator
import statistics


def g2_0(photon_beam):
    """
    calculatae g_2(0)for given photon beam

    Parameters:
        photon_beam: photon beam in question

    Returns:
        (float): g_2(0)
    """
    temp = statistics.mean(photon_beam)
    photon_beam = [x**2 for x in photon_beam]
    return (float(statistics.mean(photon_beam)) - temp) / temp**2


def generate_beam(flux:float=100, 
                  time_interval:float=0.001, 
                  total_time:float=0.5, 
                  bin_size:float=0.05):
    
    """
    generate photon counts in time intervals

    Parameters:
        flux: Average photon flux (photons per second).
        time_interval: Integration time interval for each time bin (seconds).
        total_time: Total simulation time (seconds).
        bin_size: larger time bin size for averaging counts

    Returns:
        raw_counts: Array of photon counts for small time intervals
        binned_counts: Array of photon counts averaged over larger bins  
    """
    
    num_interval = int(total_time / time_interval)
    mean_photon_count = flux * time_interval

    raw_counts = np.random.poisson(mean_photon_count, num_interval)
    raw_counts = [float(x) for x in raw_counts]

    num_interval_in_bin = int(bin_size/time_interval)
    bin_count = np.add.reduceat(raw_counts, np.arange(0, len(raw_counts), num_interval_in_bin))

    return raw_counts, bin_count


flux = 100

    )

plt.title("Simulated Photon Counts")
plt.legend()
plt.show()
