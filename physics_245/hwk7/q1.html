import numpy as np
from qutip import *
import matplotlib.pyplot as plt
        

def H_gen(omega, g, N=15):
    """
    Generate Jaynes-Cummings Hamiltonian for given parameters

    Parameters:
        omega(float): qubit frequency
        g(float): coupling strength
        N(int): hilbert space dimention for oscillator

    Returns:
         H(Qobj): the Jaynes-Cummings Hamiltonian in quesion

    """
    a = destroy(N)
    I_osc = identity(N)
    sx = sigmax()
    sz = sigmaz()
    I_qubit = identity(2)

    H = (omega/2) * tensor(sz, I_osc) + \
        omega * tensor(I_qubit, a.dag()*a + 0.5*I_osc) + \
        (g/2) * tensor(sx, I_osc) @ tensor(I_qubit, a + a.dag())
    return H


def U_gen(t, g, n):
    """
    Generate unitary time evolution operator after RWA for given parameters

    Parameters:
        t(float): a certain given time
        g(float): coupling strength
        n(int): number of photons in cavity when qubit is excited

    Returns:
        (np.array): the corresponding unitary evolution operator

    """
    Omega = g * np.sqrt(n + 1)
    c = np.cos(1/2 * Omega*t)
    s = np.sin(1/2 * Omega*t)
    
    return np.array([[c, -1j*s], 
                    [-1j*s, c]])

def prob_0_U(state):
    """
    From the given result, gather the probability of state being |0>

    Parameters:
        state(np.array): the result of U operating on initial state

    Returns:
        (float): the probability of state being |0>

    """
    return abs(state[0]**2)


def prob_0_H(result):
    """
    From the given result, gather the probability of state being |0>

    Parameters:
        result(Qobj): the result of numerical solver in QuTip

    Returns:
        probabilities(float): the list of probablities for state to be |0> in simulated 
        points int time

    """
    probabilities = []
    for state in result.states:
        rho_qubit1 = state.ptrace(0)
        prob_0 = abs(rho_qubit1[0, 0])
        probabilities.append(prob_0)

    return probabilities

    


def evolution_plot(n_init, omega, g, t_sim, sample = 1000):
    """
    plot both calculation onto the same figure

    Parameters:
        n_init(int): number of photon int the cavity at begining
        omega(float): qubit frequency
        g(float): coupling strength
        t_sim(float): the time of simulation
        sample(int): number of samples needed

    Returns:
        plt: the desired plot
    """
    N = 15
    t_list = np.linspace(0, t_sim, sample)

    # numerical simulation
    state_init = tensor(basis(2, 0), basis(N, n_init))
    H = H_gen(omega, g, N)
    result = mesolve(H, state_init, t_list)
    prob_list_H = prob_0_H(result)

    # calculation using derived U
    state_init = np.array([1, 0])
    prob_list_U = []
    for t in t_list:
        U = U_gen(t, g, n_init)
        state = U @ state_init
        prob_list_U.append(prob_0_U(state))

    plt.figure(figsize=(10, 6))
    plt.plot(t_list, prob_list_U, 'r--',label='U')
    plt.plot(t_list, prob_list_H, 'g-', label='Numerical')
    plt.legend()
    plt.xlabel('Time(s)')
    plt.ylabel('p(|0⟩)')
    plt.grid()
    return plt



omega = np.pi * 2

cases = [
    (0, omega/100, 'a'),
    (0, omega/2, 'b'),
    (10, omega/2, 'c'),
    (0, omega, 'd')
]

for n_init, g, title in cases:
    plt = evolution_plot(n_init, omega, g, 1/g * omega * 5, 1000)
    plt.title(title)
    plt.show()
