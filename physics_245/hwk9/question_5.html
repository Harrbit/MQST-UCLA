import numpy as np
import matplotlib.pyplot as plt
from qutip import basis, sigmax, sigmaz, mesolve, sigmam, mcsolve

delta = 0.0
Omega = 2.0 * np.pi
t_max = 1.0
num_points = 200
times = np.linspace(0, t_max, num_points)

sz = sigmaz()
sx = sigmax()

H = -(delta/2)*sz + (Omega/2)*sx

psi0 = basis(2,0)

c_ops = []

result = mesolve(H, psi0, times, c_ops, [basis(2,1)*basis(2,1).dag()])

rho_22 = result.expect[0]

plt.figure(figsize=(8,5))
plt.plot(times, rho_22, label=r"$\rho_{22}(t)$")
plt.xlabel("Time")
plt.ylabel(r"$\rho_{22}(t)$")
plt.title("part a")
plt.legend()
plt.grid(True)
plt.show()

gammas = [0.1*Omega, 1.0*Omega, 3.0*Omega]

t_max = 1.0
num_points = 200
times = np.linspace(0, t_max, num_points)

sm = sigmam()

fig, ax = plt.subplots(figsize=(8,5))

for Gamma in gammas:
    c_ops = [np.sqrt(Gamma)*sm]
    result = mesolve(H, psi0, times, c_ops, [basis(2,1)*basis(2,1).dag()])

ax.set_xlabel("Time")

ax.set_title("part b")

ntraj_list = [1, 10, 100]

t_max = 1.0
num_points = 200
times = np.linspace(0, t_max, num_points)


fig, axes = plt.subplots(1, 3, figsize=(15, 4))

for ax, Gamma in zip(axes, gammas):
    c_ops = [np.sqrt(Gamma) * sm]
    
    for ntraj in ntraj_list:
        result = mcsolve(
            H=-(delta/2)*sz + (Omega/2)*sx,
            state=psi0,
            tlist=times,
            c_ops=c_ops,
            e_ops=[basis(2,1)*basis(2,1).dag()],
            ntraj=ntraj,
            progress_bar=None
        )
        rho_22_avg = result.expect[0]
        ax.plot(times, rho_22_avg, label=f"{ntraj} traj")
    
    ax.set_title(rf"$\Gamma = {Gamma/Omega:.1f}\Omega$")
    ax.set_xlabel("Time")
    ax.set_ylabel(r"$\rho_{22}(t)$")
    ax.grid(True)
    ax.legend()

plt.suptitle("part c")

plt.show()
